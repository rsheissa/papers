%% *** normal ***
\newcommand{\heading}[1]{\begin{center}\large\bf #1\end{center}}

%% *** with shadowbox box (as in a window) ***
\newcommand{\Heading}[1]{%
  \begin{center}
    \large\bf
    \shadowbox{#1}%
  \end{center}
  \vspace{1ex minus 1ex}}

%% *** with doublebox box (as in a frame) ***
\newcommand{\Dheading}[1]{%
  \begin{center}
    \large\bf
    \doublebox{#1}%
  \end{center}
  \vspace{1ex minus 1ex}}

%% *** with ovalbox box ***
\newcommand{\oheading}[1]{%
  \begin{center}
    \large\bf
    \ovalbox{#1}%
  \end{center}
  \vspace{1ex minus 1ex}}

%% *** with ovalbox box  PLUS shadow (not visible in xdvi!!!) ***
\newcommand{\shoheading}[1]{%
  \begin{center}
    \large\bf
    \textshade{roundcorners}{#1}%
  \end{center}
  \vspace{1ex minus 1ex}}

%% *** with big_oval box ***
\newcommand{\Oheading}[1]{%
  \begin{center}
    \large\bf
    \Ovalbox{#1}%
  \end{center}
  \vspace{1ex minus 1ex}}
